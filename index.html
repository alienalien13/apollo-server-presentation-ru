                const typeDefs = gql`
                  type Book { title: String }
                  type Query { books: [Book] }`
            
            
                const resolvers = {
                  Query: { books: () => [{ title: 'Title' }] }
                }
            
            const server = new ApolloServer({ typeDefs, resolvers })
